#include <stdio.h>
#include <string.h>

#define EMPTY -1

int main() {
    int m, q;

    // Input size of hash table
    scanf("%d", &m);

    int table[m];

    // Initialize table
    for (int i = 0; i < m; i++) {
        table[i] = EMPTY;
    }

    // Number of operations
    scanf("%d", &q);

    char op[10];
    int key;

    while (q--) {
        scanf("%s %d", op, &key);

        // Base hash function
        int h = key % m;

        if (strcmp(op, "INSERT") == 0) {
            int i = 0;
            int idx;

            // Quadratic probing
            while (i < m) {
                idx = (h + i * i) % m;

                if (table[idx] == EMPTY) {
                    table[idx] = key;
                    break;
                }
                i++;
            }
        }

        else if (strcmp(op, "SEARCH") == 0) {
            int i = 0;
            int idx;
            int found = 0;

            // Quadratic probing search
            while (i < m) {
                idx = (h + i * i) % m;

                if (table[idx] == key) {
                    found = 1;
                    break;
                }
                if (table[idx] == EMPTY) {
                    break; // not found
                }
                i++;
            }

            if (found)
                printf("FOUND\n");
            else
                printf("NOT FOUND\n");
        }
    }

    return 0;
}
