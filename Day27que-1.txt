#include <stdio.h>
#include <stdlib.h>

// Structure of Node
struct Node {
    int data;
    struct Node* next;
};

// Function to create a new node
struct Node* createNode(int data) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = data;
    newNode->next = NULL;
    return newNode;
}

// Insert at end
struct Node* insertEnd(struct Node* head, int data) {
    struct Node* newNode = createNode(data);

    if (head == NULL)
        return newNode;

    struct Node* temp = head;
    while (temp->next != NULL)
        temp = temp->next;

    temp->next = newNode;
    return head;
}

// Find node with given value in list
struct Node* findNode(struct Node* head, int value) {
    while (head != NULL) {
        if (head->data == value)
            return head;
        head = head->next;
    }
    return NULL;
}

// Find intersection point
struct Node* findIntersection(struct Node* head1, struct Node* head2) {
    while (head1 != NULL) {
        struct Node* temp = head2;
        while (temp != NULL) {
            if (head1 == temp)  // Compare addresses
                return head1;
            temp = temp->next;
        }
        head1 = head1->next;
    }
    return NULL;
}

int main() {
    int n, m, i, value;
    struct Node *head1 = NULL, *head2 = NULL;

    // First List
    scanf("%d", &n);
    for (i = 0; i < n; i++) {
        scanf("%d", &value);
        head1 = insertEnd(head1, value);
    }

    // Second List
    scanf("%d", &m);
    struct Node* tail2 = NULL;

    for (i = 0; i < m; i++) {
        scanf("%d", &value);

        // Check if value exists in first list
        struct Node* intersectNode = findNode(head1, value);

        if (intersectNode != NULL) {
            if (head2 == NULL)
                head2 = intersectNode;
            else
                tail2->next = intersectNode;
            break;  // After intersection, stop creating new nodes
        } else {
            struct Node* newNode = createNode(value);
            if (head2 == NULL)
                head2 = newNode;
            else
                tail2->next = newNode;
            tail2 = newNode;
        }
    }

    // Find intersection
    struct Node* result = findIntersection(head1, head2);

    if (result != NULL)
        printf("%d", result->data);
    else
        printf("No Intersection");

    return 0;
}
