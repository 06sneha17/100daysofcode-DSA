#include <stdio.h>
#include <stdlib.h>

#define MAX 100

int adj[MAX][MAX];
int visited[MAX];
int recStack[MAX];
int n;

// DFS function to detect cycle
int dfs(int v) {
    visited[v] = 1;
    recStack[v] = 1;

    for (int i = 0; i < n; i++) {
        if (adj[v][i]) {
            // If not visited, recurse
            if (!visited[i]) {
                if (dfs(i))
                    return 1;
            }
            // If already in recursion stack ? cycle found
            else if (recStack[i]) {
                return 1;
            }
        }
    }

    recStack[v] = 0; // remove from recursion stack
    return 0;
}

// Function to check cycle in graph
int hasCycle() {
    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            if (dfs(i))
                return 1;
        }
    }
    return 0;
}

int main() {
    int i, j;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter adjacency matrix:\n");
    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++) {
            scanf("%d", &adj[i][j]);
        }
    }

    // Initialize arrays
    for (i = 0; i < n; i++) {
        visited[i] = 0;
        recStack[i] = 0;
    }

    if (hasCycle())
        printf("YES\n");
    else
        printf("NO\n");

    return 0;
}
