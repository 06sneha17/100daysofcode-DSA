#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    int freq[26] = {0};   // To store frequency of a-z
    int i;

    printf("Enter string: ");
    scanf("%s", str);

    // Step 1: Count frequency of each character
    for(i = 0; str[i] != '\0'; i++) {
        freq[str[i] - 'a']++;
    }

    // Step 2: Find first non-repeating character
    for(i = 0; str[i] != '\0'; i++) {
        if(freq[str[i] - 'a'] == 1) {
            printf("%c", str[i]);
            return 0;
        }
    }

    // Step 3: If no non-repeating character
    printf("$");

    return 0;
}
