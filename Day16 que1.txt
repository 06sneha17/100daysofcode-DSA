#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int arr[n];

    // Input array
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    // Sorting the array (simple bubble sort)
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }

    // Counting frequency
    int i = 0;
    while (i < n) {
        int count = 1;

        while (i < n - 1 && arr[i] == arr[i + 1]) {
            count++;
            i++;
        }

        printf("%d:%d", arr[i], count);

        if (i != n - 1) {
            printf(" ");
        }

        i++;
    }

    return 0;
}
