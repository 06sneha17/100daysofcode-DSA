#include <stdio.h>
#include <string.h>

int main() {
    int n, i, j, max = 0, count;

    printf("Enter number of votes: ");
    scanf("%d", &n);

    char names[n][50];   // to store candidate names

    printf("Enter candidate names:\n");
    for(i = 0; i < n; i++) {
        scanf("%s", names[i]);
    }

    char winner[50];

    // Find winner
    for(i = 0; i < n; i++) {
        count = 1;

        for(j = i + 1; j < n; j++) {
            if(strcmp(names[i], names[j]) == 0) {
                count++;
            }
        }

        // Update winner
        if(count > max) {
            max = count;
            strcpy(winner, names[i]);
        }
        else if(count == max) {
            // lexicographically smaller name
            if(strcmp(names[i], winner) < 0) {
                strcpy(winner, names[i]);
            }
        }
    }

    printf("%s %d\n", winner, max);

    return 0;
}
