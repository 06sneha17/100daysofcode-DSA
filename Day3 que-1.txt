#include <stdio.h>

int main() {
    int n, i, k, found = 0, comp = 0;
    int a[100];

    scanf("%d", &n);
    for(i = 0; i < n; i++)
        scanf("%d", &a[i]);

    scanf("%d", &k);

    for(i = 0; i < n; i++) {
        comp++;                 // count comparison
        if(a[i] == k) {
            printf("Found at index %d\n", i);
            found = 1;
            break;
        }
    }

    if(!found)
        printf("Not Found\n");

    printf("Comparisons = %d", comp);

    return 0;
}
