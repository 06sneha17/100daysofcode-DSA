#include <stdio.h>

int main() {
    int n, i, j;
    int count = 0;

    printf("Enter size of array: ");
    scanf("%d", &n);

    int arr[n];

    printf("Enter %d elements:\n", n);
    for(i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    // Check all possible subarrays
    for(i = 0; i < n; i++) {
        int sum = 0;
        for(j = i; j < n; j++) {
            sum = sum + arr[j];

            if(sum == 0) {
                count++;
            }
        }
    }

    printf("Number of subarrays with sum 0 = %d", count);

    return 0;
}
